CREAR REPOSITORIO------
     git init
     git status
     git add .
     git config --global user.name "nombre usuario"
     git config --global user.email "nombre-usuario@correo"
     git commit -m "comentario"
     git branch -M main
     git remote add  origin https://
     git push -u origin main

ACTUALIZAR REPOSITORIO-------
     git status
     git commit -m "comentario"
     git push -u origin main
